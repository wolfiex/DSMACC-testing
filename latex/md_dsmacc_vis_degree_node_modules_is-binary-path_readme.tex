\begin{quote}
Check if a filepath is a binary file \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install --save is-binary-path
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
var isBinaryPath = require('is-binary-path');

isBinaryPath('src/unicorn.png');
//=> true

isBinaryPath('src/unicorn.txt');
//=> false
\end{DoxyCode}
