\begin{quote}
Check if a file path is a binary file \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install is-binary-path
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
const isBinaryPath = require('is-binary-path');

isBinaryPath('source/unicorn.png');
//=> true

isBinaryPath('source/unicorn.txt');
//=> false
\end{DoxyCode}
