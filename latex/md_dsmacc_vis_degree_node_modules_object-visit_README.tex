\begin{quote}
Call a specified method on each value in the given object. \end{quote}


\subsection*{Install}

Install with \href{https://www.npmjs.com/}{\tt npm}\+:


\begin{DoxyCode}
$ npm install --save object-visit
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
var visit = require('object-visit');

var ctx = \{
  data: \{\},
  set: function (key, value) \{
    if (typeof key === 'object') \{
      visit(ctx, 'set', key);
    \} else \{
      ctx.data[key] = value;
    \}
  \}
\};

ctx.set('a', 'a');
ctx.set('b', 'b');
ctx.set('c', 'c');
ctx.set(\{d: \{e: 'f'\}\});

console.log(ctx.data);
//=> \{a: 'a', b: 'b', c: 'c', d: \{ e: 'f' \}\};
\end{DoxyCode}
