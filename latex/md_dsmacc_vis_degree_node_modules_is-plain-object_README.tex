\begin{quote}
Returns true if an object was created by the {\ttfamily Object} constructor. \end{quote}


\subsection*{Install}

Install with \href{https://www.npmjs.com/}{\tt npm}\+:


\begin{DoxyCode}
$ npm install --save is-plain-object
\end{DoxyCode}


Use \href{https://github.com/jonschlinkert/isobject}{\tt isobject} if you only want to check if the value is an object and not an array or null.

\subsection*{Usage}


\begin{DoxyCode}
var isPlainObject = require('is-plain-object');
\end{DoxyCode}


{\bfseries true} when created by the {\ttfamily Object} constructor.


\begin{DoxyCode}
isPlainObject(Object.create(\{\}));
//=> true
isPlainObject(Object.create(Object.prototype));
//=> true
isPlainObject(\{foo: 'bar'\});
//=> true
isPlainObject(\{\});
//=> true
\end{DoxyCode}


{\bfseries false} when not created by the {\ttfamily Object} constructor.


\begin{DoxyCode}
isPlainObject(1);
//=> false
isPlainObject(['foo', 'bar']);
//=> false
isPlainObject([]);
//=> false
isPlainObject(new Foo);
//=> false
isPlainObject(null);
//=> false
isPlainObject(Object.create(null));
//=> false
\end{DoxyCode}
