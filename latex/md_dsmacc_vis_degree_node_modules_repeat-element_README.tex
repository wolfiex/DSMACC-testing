\begin{quote}
Create an array by repeating the given value n times. \end{quote}


Please consider following this project\textquotesingle{}s author, \href{https://github.com/jonschlinkert}{\tt Jon Schlinkert}, and consider starring the project to show your \+:heart\+: and support.

\subsection*{Install}

Install with \href{https://www.npmjs.com/}{\tt npm}\+:


\begin{DoxyCode}
$ npm install --save repeat-element
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
const repeat = require('repeat-element');

repeat('a', 5);
//=> ['a', 'a', 'a', 'a', 'a']

repeat('a', 1);
//=> ['a']

repeat('a', 0);
//=> []

repeat(null, 5)
//» [ null, null, null, null, null ]

repeat(\{some: 'object'\}, 5)
//» [ \{ some: 'object' \},
//    \{ some: 'object' \},
//    \{ some: 'object' \},
//    \{ some: 'object' \},
//    \{ some: 'object' \} ]

repeat(5, 5)
//» [ 5, 5, 5, 5, 5 ]
\end{DoxyCode}
