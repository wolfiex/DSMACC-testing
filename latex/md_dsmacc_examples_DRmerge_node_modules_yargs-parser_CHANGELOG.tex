// => \{ \_: [], flag: true \}

parse('--no-flag', \{boolean: ['flag']\});
// => \{ \_: [], flag: false \}

parse('', \{boolean: ['flag']\});
// => \{ \_: [] \} => flag not set similarly to other option type
\end{DoxyCode}
