\subsection*{Usage}


\begin{DoxyCode}
const camelCase = require('camelcase');

camelCase('foo-bar');
//=> 'fooBar'

camelCase('foo\_bar');
//=> 'fooBar'

camelCase('Foo-Bar');
//=> 'fooBar'

camelCase('Foo-Bar', \{pascalCase: true\});
//=> 'FooBar'

camelCase('--foo.bar', \{pascalCase: false\});
//=> 'fooBar'

camelCase('foo bar');
//=> 'fooBar'

console.log(process.argv[3]);
//=> '--foo-bar'
camelCase(process.argv[3]);
//=> 'fooBar'

camelCase(['foo', 'bar']);
//=> 'fooBar'

camelCase(['\_\_foo\_\_', '--bar'], \{pascalCase: true\});
//=> 'FooBar'
\end{DoxyCode}
