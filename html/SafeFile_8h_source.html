104 struct BinaryHelper {

106  {
107  return fwrite(reinterpret_cast<const char*>(&value), sizeof(value), 1, file);
108  }

110  static size_t read(T& value, ifstream& ifstream)

118 template<>
119 struct BinaryHelper<std::string> {
120  static size_t write(std::string value, std::FILE* file)
121  {
122  unsigned short stringLength = static_cast<unsigned short>(value.length());
123  fwrite(&stringLength, sizeof(stringLength), 1, file);
124  return fwrite(value.c_str(), sizeof(char), value.length(), file);
125  }
126 
127  static size_t read(std::string& value, ifstream& ifstream)
128  {
129  unsigned short stringLength;
130  size_t size = sizeof(stringLength);
131  ifstream.read(reinterpret_cast<char*>(&stringLength), size);
132  if (stringLength > 0)
133  {
134  char cstr[stringLength];
135  ifstream.read(cstr, stringLength);
136  std::string str(cstr, stringLength);
137  value.swap(str);
138  }
139  return size + sizeof(char) * stringLength;
140  }
141 };
