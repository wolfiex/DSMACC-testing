89 inline
90 void InfomapGreedySpecialized<FlowDirectedWithTeleportation>::addTeleportationFlowOnLeafNodes()
91 {
92  double alpha = m_config.teleportationProbability;
93  double beta = 1.0 - alpha;
94  for (TreeData::leafIterator it(m_treeData.begin_leaf()), itEnd(m_treeData.end_leaf());
95  it != itEnd; ++it)
96  {
97  NodeType& node = getNode(**it);
98  // Don't let self-teleportation add to the enter/exit flow (i.e. multiply with (1.0 - node.data.teleportWeight))
99  node.data.exitFlow += (alpha * node.data.flow + beta * node.data.danglingFlow) * (1.0 - node.data.teleportWeight);
100  node.data.enterFlow += (alpha * (1.0 - node.data.flow) + beta * (m_sumDanglingFlow - node.data.danglingFlow)) * node.data.teleportWeight;
101  }
102 }

104 template<>
105 inline
106 void InfomapGreedySpecialized<FlowDirectedWithTeleportation>::addTeleportationFlowOnModules()
107 {
108  double alpha = m_config.teleportationProbability;
109  double beta = 1.0 - alpha;
110 
111  for (NodeBase::pre_depth_first_iterator it(root()); !it.isEnd(); ++it)
112  {
113  NodeType& node = getNode(*it);
114  if (!node.isLeaf())
115  {
116  // Don't code self-teleportation
117  node.data.enterFlow += (alpha * (1.0 - node.data.flow) + beta * (m_sumDanglingFlow - node.data.danglingFlow)) * node.data.teleportWeight;
118  node.data.exitFlow += (alpha * node.data.flow + beta * node.data.danglingFlow) * (1.0 - node.data.teleportWeight);
119  }
120  }

123 template<typename FlowType>
124 inline
125 void InfomapGreedySpecialized<FlowType>::initEnterExitFlow()
126 {
127  for (TreeData::leafIterator it(Super::m_treeData.begin_leaf()), itEnd(Super::m_treeData.end_leaf());
128  it != itEnd; ++it)
129  {
130  NodeBase& node = **it;
131  for (NodeBase::edge_iterator edgeIt(node.begin_outEdge()), edgeEnd(node.end_outEdge());
132  edgeIt != edgeEnd; ++edgeIt)
133  {
134  EdgeType& edge = **edgeIt;
135  // Possible self-links should not add to enter and exit flow in its enclosing module
136  if (edge.source != edge.target)
137  {
138  // For undirected links, this automatically adds to both direction (as enterFlow = &exitFlow)
139  Super::getNode(edge.source).data.exitFlow += edge.data.flow;
140  Super::getNode(edge.target).data.enterFlow += edge.data.flow;
141  }
142  }
143  }

150 template<>
151 inline
152 void InfomapGreedySpecialized<FlowDirectedWithTeleportation>::initEnterExitFlow()
153 {
154  for (TreeData::leafIterator it(m_treeData.begin_leaf()), itEnd(m_treeData.end_leaf());
155  it != itEnd; ++it)
156  {
157  NodeBase& node = **it;
158  FlowType& data = getNode(node).data;
159  // Also store the flow to use for teleportation as the flow can be transformed
160  data.teleportSourceFlow = data.flow;
161  if (node.isDangling())
162  {
163  m_sumDanglingFlow += data.flow;
164  data.danglingFlow = data.flow;
165  }
166  else
167  {
168  for (NodeBase::edge_iterator edgeIt(node.begin_outEdge()), edgeEnd(node.end_outEdge());
169  edgeIt != edgeEnd; ++edgeIt)
170  {
171  EdgeType& edge = **edgeIt;
172  // Possible self-links should not add to enter and exit flow in its enclosing module
173  if (edge.source != edge.target)
174  {
175  getNode(edge.source).data.exitFlow += edge.data.flow;
176  getNode(edge.target).data.enterFlow += edge.data.flow;
177  }
178  }
179  }
180  }
181 
182  addTeleportationFlowOnLeafNodes();
183 }

190  double beta = 1.0 - alpha;
191  FlowType& oldModuleFlowData = m_moduleFlowData[oldModuleDeltaFlow.module];
192  oldModuleDeltaFlow.deltaExit += (alpha*nodeToMove.data.teleportSourceFlow + beta*nodeToMove.data.danglingFlow) * (oldModuleFlowData.teleportWeight - nodeToMove.data.teleportWeight);
193  oldModuleDeltaFlow.deltaEnter += (alpha*(oldModuleFlowData.teleportSourceFlow - nodeToMove.data.teleportSourceFlow) +
194  beta*(oldModuleFlowData.danglingFlow - nodeToMove.data.danglingFlow)) * nodeToMove.data.teleportWeight;

240 template<typename FlowType>
241 inline
242 double InfomapGreedySpecialized<FlowType>::getDeltaCodelengthOnMovingNode(NodeType& current,
243  DeltaFlow& oldModuleDelta, DeltaFlow& newModuleDelta)
244 {
245  using infomath::plogp;
246  std::vector<FlowType>& moduleFlowData = Super::m_moduleFlowData;
247  unsigned int oldModule = oldModuleDelta.module;
248  unsigned int newModule = newModuleDelta.module;
249  double deltaEnterExitOldModule = oldModuleDelta.deltaEnter + oldModuleDelta.deltaExit;
250  double deltaEnterExitNewModule = newModuleDelta.deltaEnter + newModuleDelta.deltaExit;
251 
252  double delta_enter = plogp(Super::enterFlow + deltaEnterExitOldModule - deltaEnterExitNewModule) - Super::enterFlow_log_enterFlow;
253 
254  double delta_enter_log_enter = \
255  - plogp(moduleFlowData[oldModule].enterFlow) \
256  - plogp(moduleFlowData[newModule].enterFlow) \
257  + plogp(moduleFlowData[oldModule].enterFlow - current.data.enterFlow + deltaEnterExitOldModule) \
258  + plogp(moduleFlowData[newModule].enterFlow + current.data.enterFlow - deltaEnterExitNewModule);
259 
260  double delta_exit_log_exit = \
261  - plogp(moduleFlowData[oldModule].exitFlow) \
262  - plogp(moduleFlowData[newModule].exitFlow) \
263  + plogp(moduleFlowData[oldModule].exitFlow - current.data.exitFlow + deltaEnterExitOldModule) \
264  + plogp(moduleFlowData[newModule].exitFlow + current.data.exitFlow - deltaEnterExitNewModule);
265 
266  double delta_flow_log_flow = \
267  - plogp(moduleFlowData[oldModule].exitFlow + moduleFlowData[oldModule].flow) \
268  - plogp(moduleFlowData[newModule].exitFlow + moduleFlowData[newModule].flow) \
269  + plogp(moduleFlowData[oldModule].exitFlow + moduleFlowData[oldModule].flow \
270  - current.data.exitFlow - current.data.flow + deltaEnterExitOldModule) \
271  + plogp(moduleFlowData[newModule].exitFlow + moduleFlowData[newModule].flow \
272  + current.data.exitFlow + current.data.flow - deltaEnterExitNewModule);
273 
274  double deltaL = delta_enter - delta_enter_log_enter - delta_exit_log_exit + delta_flow_log_flow;
275  return deltaL;
276 }

363  Super::flow_log_flow += \
364  plogp(moduleFlowData[oldModule].exitFlow + moduleFlowData[oldModule].flow) + \
365  plogp(moduleFlowData[newModule].exitFlow + moduleFlowData[newModule].flow);
