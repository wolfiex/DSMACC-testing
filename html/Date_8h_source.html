40 class ElapsedTime
41 {
42 public:
43  enum InSeconds
44  {
45  SECONDS_IN_A_MINUTE = 60,
46  SECONDS_IN_AN_HOUR = 3600,
47  SECONDS_IN_A_DAY = 86400 // 3600*24
48  };
49 
50  ElapsedTime(double elapsedTime) : m_elapsedTime(elapsedTime) {}
51  ~ElapsedTime() {}
52 
53  double getSeconds() const { return m_elapsedTime; }
54  double getMinutes() const { return m_elapsedTime / 60; }
55  double getHours() const { return m_elapsedTime / 3600; }
56  double getDays() const { return m_elapsedTime / 86400; }
57 
58 
59  friend std::ostream& operator<<(std::ostream& out, const ElapsedTime& elapsedTime)
60  {
61  unsigned int temp = static_cast<unsigned int>(std::floor(elapsedTime.getSeconds()));
62  if (temp > 60)
63  {
64  if (temp > 3600)
65  {
66  if (temp > 86400)
67  {
68  out << temp / 86400 << "d ";
69  temp %= 86400;
70  }
71  out << temp / 3600 << "h ";
72  temp %= 3600;
73  }
74  out << temp / 60 << "m ";
75  temp %= 60;
76  out << temp << "s";
77  }
78  else
79  {
80  out << temp << "s";
81  }
82  return out;
83  }
84 private:
85  double m_elapsedTime;
86 };

88 class Date

99 
100  friend std::ostream& operator<<(std::ostream& out, const Date& date)
101  {
102  struct std::tm t = *localtime(&date.m_timeOfCreation);
103  return out << "[" <<
104  (t.tm_year+1900) <<
105  (t.tm_mon < 9 ? "-0" : "-") <<
106  (t.tm_mon+1) <<
107  (t.tm_mday < 10 ? "-0" : "-") <<
108  t.tm_mday <<
109  (t.tm_hour < 10 ? " 0" : " ") <<
110  t.tm_hour <<
111  (t.tm_min < 10 ? ":0" : ":") <<
112  t.tm_min <<
113  (t.tm_sec < 10 ? ":0" : ":") <<
114  t.tm_sec << "]";
115  }
116 
117  ElapsedTime operator-(const Date& date)
118  {
119  return ElapsedTime(difftime(m_timeOfCreation, date.m_timeOfCreation));
120  }
121 
122 private:
123  std::time_t m_timeOfCreation;
